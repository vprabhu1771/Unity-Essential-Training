# Import Sprites

Assets -> Sprites -> Import

# Go To Inspector

Sprite Mode -> Multiple

Advanced -> Filter Mode -> Point (no filter)

Advanced -> Default -> Compression -> None

Click -> Apply

Click -> Sprite Editor
